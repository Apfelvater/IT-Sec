#!/usr/bin/env python2
import os
from Crypto.Cipher import AES
from Crypto.Util import Counter


def encrypt(msg):
    nonce = os.urandom(16)
    key = os.urandom(32)
    aes = AES.new(key, AES.MODE_CTR, counter=lambda: nonce)
    return aes.encrypt(msg)
